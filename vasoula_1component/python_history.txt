def find_differences(query_seq, subject_seq):
    # Initialize the pairwise aligner
    aligner = Align.PairwiseAligner()
    # Perform pairwise alignment
    alignments = aligner.align(query_seq, subject_seq)
    # Get the first alignment (assuming there's only one)
    alignment = alignments[0]
    # Extract aligned sequences
    query_aligned = alignment.query
    subject_aligned = alignment.target
    # Find the positions where the sequences differ
    differences = []
    for i, (q, s) in enumerate(zip(query_aligned, subject_aligned)):
        if q != s:
            differences.append(f"At position {i+1}: {q} -> {s}")
    return differences

    gene_pattern = re.compile(r'\[gene=([^\]]+)\]')
    genes = set()
    # Open the file based on whether it's gzipped or not
    with gzip.open(file_path, 'rt') as file:
        for line in file:
            if line.startswith('>'):
                match = gene_pattern.search(line)
                if match:
                    genes.add(match.group(1))
    return genes
